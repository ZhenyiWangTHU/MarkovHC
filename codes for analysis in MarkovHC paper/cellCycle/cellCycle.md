pathLength <- matrix(0,3,3)
level <- 17
for(i in 1:3){
  for(j in 1:3){
    MarkovHCPath <- findTransitionPath(MarkovObject = MarkovHC_result,
                                       level = level,

    if(length(MarkovHCPath)>0){pathLength[i,j] <- MarkovHCPath[[3]]}else{
